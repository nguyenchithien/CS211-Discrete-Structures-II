        \item	Length 5 means 1x, 3x, or 5x ones.
					\\ one 1, four 0:	$r = 1$, $n-r = 4$, $n-1=4, n=5$, $C(5,1) = 5$
					\\ three 1, two 0:	$r = 3$, $n-r = 2$, $n-3=2, n=5$, $C(5,3) = 10$
					\\ five 1, zero 0:	$r = 5$, $n-r = 0$, $n-5=0, n=5$, $C(5,5) = 1$
					\\ $5 + 10 + 1 = 16$
